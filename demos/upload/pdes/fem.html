def phi(x, t, i, k):
    """
    A basis function
    
    x : float or array
        evaluation point
    
    t : array
        mesh nodes
        
    i : int
        basis id
        
    k : int
        basis degree
    """
    b = BSpline.basis_element(t[i:i+k+2], extrapolate=False)
    s = np.nan_to_num(b(x))
    return s

def dphi(x, t, i, k):
    if np.abs(t[i+k]  -t[i]) < 1e-14:
        p1 = 0*x
    else:
        p1 = phi(x, t, i,   k-1) / (t[i+k]  -t[i])
    if np.abs(t[i+k+1]-t[i+1]):
        p2 = 0*x
    else:
        p2 = phi(x, t, i+1, k-1) / (t[i+k+1]-t[i+1])
    ds = k*(p1 - p2)
    return ds
