def power_method(A, x, maxiter):
    eigs = np.zeros(maxiter)
    eigv = np.zeros((len(x), maxiter))

    for i in range(maxiter):
        y = A @ x
        ynorm = np.linalg.norm(y,np.inf)
        x = y / ynorm
        
        eigs[i] = ynorm
        eigv[:,i] = x
        
    return x, eigs, eigv
